function SkillUpExpress(sExpress,A,B,C,D,E,F,G,X)
	sExpress = string.gsub(sExpress,"A",tostring(A));
	sExpress = string.gsub(sExpress,"B",tostring(B));
	sExpress = string.gsub(sExpress,"Control",tostring(C));
	sExpress = string.gsub(sExpress,"Charm",tostring(D));
	sExpress = string.gsub(sExpress,"C",tostring(E));
	sExpress = string.gsub(sExpress,"Force",tostring(F));
	sExpress = string.gsub(sExpress,"Polity",tostring(G));
	sExpress = string.gsub(sExpress,"X",tostring(X));
	return loadstring("return " .. sExpress)();
end
